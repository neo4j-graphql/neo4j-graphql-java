enum Gender { female, male }
type Person {
  id : ID!
  name: String
  age: Int
  height: Float
  fun: Boolean
  gender: Gender
  company: Company @relation(name:"WORKS_AT")
  location: _Neo4jPoint
}

  _id: ID
  name: String
  employees: [Person] @relation(name:"WORKS_AT", direction: IN)
}
----

[source,graphql]
----

----

.GraphQL-Response
[source,json,response=true,ignore-order]
----
{
  "person" : [ {
    "name" : "Joe"
  } ]
}

[source,json]
----

  "filterPersonGender" : "male"
