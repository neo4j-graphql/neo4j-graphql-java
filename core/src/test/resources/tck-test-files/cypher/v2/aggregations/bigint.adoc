= Cypher Aggregations BigInt

== Source schema

[source,graphql,schema=true]
----
type File {
  size: BigInt!
}
----
== Average

.GraphQL-Query
[source,graphql]
----
{
  filesAggregate {
    size {
      average
    }
  }
}
----

.Expected Cypher params
[source,json]
----
{}
----

.Expected Cypher output
[source,cypher]
----
CALL {
    MATCH (this:File)
    RETURN { average: avg(this.size) } AS var0
}
RETURN { size: var0 }
----

'''
