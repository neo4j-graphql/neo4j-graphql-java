= Cypher Aggregations Count

== Source schema

[source,graphql,schema=true]
----
type Movie {
  title: String!
}
----
== Count with WHERE

.GraphQL-Query
[source,graphql]
----
{
  moviesAggregate(where: {title: "some-title"}) {
    count
  }
}
----

.Expected Cypher params
[source,json]
----
{
  "param0" : "some-title"
}
----

.Expected Cypher output
[source,cypher]
----
CALL {
	MATCH (this:Movie)
	WHERE this.title = $param0
	RETURN count(this) AS var0
}
RETURN {
	count: var0
}
----
