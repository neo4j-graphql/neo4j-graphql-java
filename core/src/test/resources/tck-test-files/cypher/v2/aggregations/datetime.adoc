= Cypher Aggregations DateTime

== Source schema

[source,graphql,schema=true]
----
type Movie {
  createdAt: DateTime!
}
----
== Max

.GraphQL-Query
[source,graphql]
----
{
  moviesAggregate {
    createdAt {
      max
    }
  }
}
----

.Expected Cypher params
[source,json]
----
{ }
----

.Expected Cypher output
[source,cypher]
----
CALL {
	MATCH (this:Movie)
	RETURN {
		max: apoc.date.convertFormat(toString(max(this.createdAt)), 'iso_zoned_date_time', 'iso_offset_date_time')
	} AS var0
}
RETURN {
	createdAt: var0
}
----

'''
