= Cypher Aggregations Duration

== Source schema

[source,graphql,schema=true]
----
type Movie {
  screenTime: Duration!
}
----
== Max

.GraphQL-Query
[source,graphql]
----
{
  moviesAggregate {
    screenTime {
      max
    }
  }
}
----

.Expected Cypher params
[source,json]
----
{ }
----

.Expected Cypher output
[source,cypher]
----
CALL {
	MATCH (this:Movie)
	RETURN {
		max: max(this.screenTime)
	} AS var0
}
RETURN {
	screenTime: var0
}
----

'''
