= Cypher Aggregations Float

== Source schema

[source,graphql,schema=true]
----
type Movie {
  actorCount: Float!
}
----
== Average

.GraphQL-Query
[source,graphql]
----
{
  moviesAggregate {
    actorCount {
      average
    }
  }
}
----

.Expected Cypher params
[source,json]
----
{ }
----

.Expected Cypher output
[source,cypher]
----
CALL {
	MATCH (this:Movie)
	RETURN {
		average: avg(this.actorCount)
	} AS var0
}
RETURN {
	actorCount: var0
}
----

'''
