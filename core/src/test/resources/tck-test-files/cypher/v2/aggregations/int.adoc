= Cypher Aggregations Int

== Source schema

[source,graphql,schema=true]
----
type Movie {
  imdbRating: Int!
}
----
== Average

.GraphQL-Query
[source,graphql]
----
{
  moviesAggregate {
    imdbRating {
      average
    }
  }
}
----

.Expected Cypher params
[source,json]
----
{ }
----

.Expected Cypher output
[source,cypher]
----
CALL {
	MATCH (this:Movie)
	RETURN {
		average: avg(this.imdbRating)
	} AS var0
}
RETURN {
	imdbRating: var0
}
----

'''
