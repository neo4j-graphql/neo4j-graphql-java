= Cypher Aggregations LocalDateTime

== Source schema

[source,graphql,schema=true]
----
type Movie {
  createdAt: LocalDateTime!
}
----
== Max

.GraphQL-Query
[source,graphql]
----
{
  moviesAggregate {
    createdAt {
      max
    }
  }
}
----

.Expected Cypher params
[source,json]
----
{ }
----

.Expected Cypher output
[source,cypher]
----
CALL {
	MATCH (this:Movie)
	RETURN {
		max: max(this.createdAt)
	} AS var0
}
RETURN {
	createdAt: var0
}
----

'''
