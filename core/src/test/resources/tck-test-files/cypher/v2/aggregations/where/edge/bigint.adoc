== Source schema

type User {
  someBigInt: BigInt
}

type Post {
  content: String!
  likes: [User!]! @relationship(type: "LIKES", direction: IN, properties: "Likes")
}

interface Likes @relationshipProperties {
  someBigInt: BigInt
  someBigIntAlias: BigInt @alias(property: "_someBigIntAlias")
}
----
