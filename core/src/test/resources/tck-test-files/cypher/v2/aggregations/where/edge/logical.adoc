= Cypher Aggregations where edge with Logical AND + OR + NOT

== Source schema

[source,graphql,schema=true]
----
type User {
  name: String
}

type Post {
  content: String!
  likes: [User!]! @relationship(type: "LIKES", direction: IN, properties: "Likes")
}

type Likes @relationshipProperties {
  someFloat: Float
}
----
== AND

.GraphQL-Query
[source,graphql]
----
{
  posts(
    where: {likesAggregate: {edge: {AND: [{someFloat_SUM_EQUAL: 10}, {someFloat_SUM_EQUAL: 11}]}}}
  ) {
    content
  }
}
----

.Expected Cypher params
[source,json]
----
{
  "param0": 10,
  "param1": 11
}
----

.Expected Cypher output
[source,cypher]
----
MATCH (this:Post)
CALL {
    WITH this
    MATCH (this)<-[this0:LIKES]-(this1:User)
    RETURN (sum(this0.someFloat) = $param0 AND sum(this0.someFloat) = $param1) AS var2
}
WITH *
WHERE var2 = true
RETURN this { .content } AS this
----

'''
