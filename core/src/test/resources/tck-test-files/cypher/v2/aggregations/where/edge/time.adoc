= Cypher Aggregations where edge with Time

== Source schema

[source,graphql,schema=true]
----
type User {
  name: String
}

type Post {
  content: String!
  likes: [User!]! @relationship(type: "LIKES", direction: IN, properties: "Likes")
}

interface Likes {
  someTime: Time
  someTimeAlias: Time @alias(property: "_someTimeAlias")
}
----

== Configuration

.Configuration
[source,json,schema-config=true]
----
{
  "enableRegex": true
}
----
== EQUAL

.GraphQL-Query
[source,graphql]
----
{
  posts(where: {likesAggregate: {edge: {someTime_EQUAL: "12:00:00"}}}) {
    content
  }
}
----

.Expected Cypher params
[source,json]
----
{
  "param0": {
    "hour": 12,
    "minute": 0,
    "second": 0,
    "nanosecond": 0,
    "timeZoneOffsetSeconds": 0
  }
}
----

.Expected Cypher output
[source,cypher]
----
MATCH (this:Post)
CALL {
	WITH this
	MATCH (this1:User)-[this0:LIKES]->(this)
	RETURN any(var2 IN collect(this0.someTime) WHERE var2 = $param0) AS var3
}
WITH *
WHERE var3 = true
RETURN this {
	.content
} AS this
----

'''
