= Cypher Aggregations where node with BigInt

== Source schema

[source,graphql,schema=true]
----
type User {
  someBigInt: BigInt
  someBigIntAlias: BigInt @alias(property: "_someBigIntAlias")
}

type Post {
  content: String!
  likes: [User!]! @relationship(type: "LIKES", direction: IN)
}
----
