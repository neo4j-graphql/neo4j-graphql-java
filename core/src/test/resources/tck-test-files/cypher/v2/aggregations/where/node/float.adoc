= Cypher Aggregations where node with Float

== Source schema

[source,graphql,schema=true]
----
type User {
  someFloat: Float
  someFloatAlias: Float @alias(property: "_someFloatAlias")
}

type Post {
  content: String!
  likes: [User!]! @relationship(type: "LIKES", direction: IN)
}
----
