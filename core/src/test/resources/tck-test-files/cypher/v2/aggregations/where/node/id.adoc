= Cypher Aggregations where node with ID

== Source schema

[source,graphql,schema=true]
----
type User {
  id: ID
  someIdAlias: ID @alias(property: "_someIdAlias")
  name: String!
}

type Post {
  content: String!
  likes: [User!]! @relationship(type: "LIKES", direction: IN)
}
----

== Configuration

.Configuration
[source,json,schema-config=true]
----
{
  "enableRegex": true
}
----
== EQUAL

.GraphQL-Query
[source,graphql]
----
{
  posts(where: {likesAggregate: {node: {id_EQUAL: "10"}}}) {
    content
  }
}
----

.Expected Cypher params
[source,json]
----
{
  "param0": "10"
}
----

.Expected Cypher output
[source,cypher]
----
MATCH (this:Post)
CALL {
	WITH this
	MATCH (this1:User)-[this0:LIKES]->(this)
	RETURN any(var2 IN collect(this1.id) WHERE var2 = $param0) AS var3
}
WITH *
WHERE var3 = true
RETURN this {
	.content
} AS this
----

'''
