= Cypher Aggregations where node with Int

== Source schema

[source,graphql,schema=true]
----
type User {
  someInt: Int
  someIntAlias: Int @alias(property: "_someIntAlias")
}

type Post {
  content: String!
  likes: [User!]! @relationship(type: "LIKES", direction: IN)
}
----
