== Source schema

[source,graphql,schema=true]
----
type Source @authorization(validate: [{when: [BEFORE], operations: [CREATE_RELATIONSHIP], where: {node: {id: "$jwt.sub"}}}]) {
  id: ID!
  targets: [Target!]! @relationship(type: "HAS_TARGET", direction: OUT)
}

type Target {
  id: ID!
}
----
