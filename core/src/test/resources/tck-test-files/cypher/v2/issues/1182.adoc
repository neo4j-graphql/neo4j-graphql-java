== Source schema

[source,graphql,schema=true]
----
type Movie {
  id: ID! @id
  title: String!
  actors: [Actor!]! @relationship(type: "ACTED_IN", direction: IN)
}

type Actor {
  id: ID! @id
  name: String!
  dob: DateTime!
  homeAddress: Point!
}
----
