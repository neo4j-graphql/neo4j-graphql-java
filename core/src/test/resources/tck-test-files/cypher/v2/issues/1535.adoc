== Source schema

[source,graphql,schema=true]
----
type Tenant {
  id: ID! @id @unique
  name: String!
  events: [Event!]! @relationship(type: "HOSTED_BY", direction: IN)
  fooBars: [FooBar!]! @relationship(type: "HAS_FOOBARS", direction: OUT)
}

interface Event {
  id: ID!
  title: String
  beginsAt: DateTime!
}

type Screening implements Event {
  id: ID! @id @unique
  title: String
  beginsAt: DateTime!
}

type Booking implements Event {
  id: ID!
  title: String
  beginsAt: DateTime!
  duration: Int!
}

type FooBar {
  id: ID! @id @unique
  name: String!
}
----
