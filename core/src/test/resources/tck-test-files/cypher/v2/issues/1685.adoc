== Source schema

[source,graphql,schema=true]
----
interface Production {
  id: ID
  title: String
}

type Movie implements Production {
  id: ID
  title: String
  actorCount: Int
  genres: [Genre!]! @relationship(type: "HAS_GENRE", direction: OUT)
}

type Genre {
  name: String
  movies: [Production!]! @relationship(type: "HAS_GENRE", direction: IN)
}
----

	WITH this
	CALL {
		WITH this
