= Cypher directive

== Source schema

[source,graphql,schema=true]
----
type Actor {
  name: String
  movies(title: String): [Movie] @cypher(statement: """
  MATCH (m:Movie {title: $title})
  RETURN m
  """, columnName: "m")
}

type Movie {
  id: ID
  title: String
  actors: [Actor!]! @relationship(type: "ACTED_IN", direction: IN)
}
----
== Nested Connection

.GraphQL-Query
[source,graphql]
----
{
  actors {
    movies {
      actorsConnection {

      }
    }
  }
}
----

[source,json]

{ }
