= Cypher Date

== Source schema

[source,graphql,schema=true]
----
type Movie {
  id: ID
  date: Date
}
----

== GTE Read

.GraphQL-Query
[source,graphql]
----
{
  movies(where: {date_GTE: "1980-04-08"}) {
    date
  }
}
----

.Expected Cypher params
[source,json]
----
{
  "param0" : "1980-04-08"
}
----

.Expected Cypher output
[source,cypher]
----
MATCH (this:Movie)
WHERE this.date >= $param0
RETURN this {
	.date
} AS this
----
