= Cypher Duration

== Setup

.Schema
[source,graphql,schema=true]
----
type Movie {
  id: ID
  duration: Duration
}
----

== Simple Read

.GraphQL-Query
[source,graphql]
----
{
  movies(where: {duration: "P1Y"}) {
    duration
  }
}
----

.Expected Cypher params
[source,json]
----
{
  "param0" : "P1Y"
}
----

.Expected Cypher output
[source,cypher]
----
MATCH (this:Movie)
WHERE (datetime() + this.duration) = (datetime() + $param0)
RETURN this {
	.duration
} AS this
----

'''
