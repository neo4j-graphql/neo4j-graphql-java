= Cypher LocalTime

== Source schema

[source,graphql,schema=true]
----
type Movie {
  id: ID
  time: LocalTime
}
----

== GTE Read

.GraphQL-Query
[source,graphql]
----
{
  movies(where: {time_GTE: "13:45:33.250"}) {
    time
  }
}
----

.Expected Cypher params
[source,json]
----
{
  "param0" : "13:45:33.250"
}
----

.Expected Cypher output
[source,cypher]
----
MATCH (this:Movie)
WHERE this.time >= $param0
RETURN this {
	.time
} AS this
----

'''
