= Cypher Time

== Setup

.Schema
[source,graphql,schema=true]
----
type Movie {
  id: ID
  time: Time
}
----

== Simple Read

.GraphQL-Query
[source,graphql,request=true]
----
{
  movies(where: {time: "12:00:00"}) {
    time
  }
}
----

.Expected Cypher params
[source,json]
----
{
  "param0" : "12:00Z"
}
----

.Expected Cypher output
[source,cypher]
----
MATCH (this:Movie)
WHERE this.time = $param0
RETURN this {
	.time
} AS this
----

== GTE Read

.GraphQL-Query
[source,graphql,request=true]
----
{
  movies(where: {time_GTE: "13:45:33.250"}) {
    time
  }
}
----

.Expected Cypher params
[source,json]
----
{
  "param0" : "13:45:33.250Z"
}
----

.Expected Cypher output
[source,cypher]
----
MATCH (this:Movie)
WHERE this.time >= $param0
RETURN this {
	.time
} AS this
----
