== Source schema

type Actor {
  name: String
  totalScreenTime: Int! @cypher(statement: """
  MATCH (this)-[r:ACTED_IN]->(:Movie)
  RETURN sum(r.screenTime) as result
  """, columnName: "result")
}

type Movie {
  id: ID
  actors(title: String): [Actor] @cypher(statement: """
  MATCH (a:Actor {title: $title})
  RETURN a
  LIMIT 1
  """, columnName: "a")
}
----
