== Source schema

[source,graphql,schema=true]
----
type PersonOne {
  name: String
}

type PersonTwo {
  nameTwo: String
}

union Person = PersonOne | PersonTwo

type Movie {
  id: ID
  actors: [Person!]! @relationship(type: "ACTED_IN", direction: IN, nestedOperations: [DELETE])
}
----
