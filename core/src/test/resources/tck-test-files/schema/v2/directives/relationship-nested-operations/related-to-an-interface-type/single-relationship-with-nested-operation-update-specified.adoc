== Source schema

[source,graphql,schema=true]
----
interface Person {
  name: String
}

type PersonOne implements Person {
  name: String
  someExtraProp: [Int!]!
}

type PersonTwo implements Person {
  name: String
}

type Movie {
  id: ID
  actors: [Person!]! @relationship(type: "ACTED_IN", direction: IN, nestedOperations: [UPDATE])
}
----
