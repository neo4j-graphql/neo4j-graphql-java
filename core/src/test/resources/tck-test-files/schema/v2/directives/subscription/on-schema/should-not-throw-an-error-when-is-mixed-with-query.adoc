= @subscription directive -> on SCHEMA -> should not throw an Error when is mixed with @query

== Source schema

[source,graphql,schema=true]
----
type Actor {
  username: String!
  password: String!
  actedIn: [Movie!]! @relationship(type: "ACTED_IN", direction: IN)
}

type Movie {
  title: String
  actors: [Actor!]! @relationship(type: "ACTED_IN", direction: OUT)
}

extend schema @query(read: true) @subscription(events: []) {
  query: Query
  mutation: Mutation
}
----
