== Source schema

[source,graphql,schema=true]
----
type PersonOne {
  id: ID! @id @unique
  name: String
}

type PersonTwo {
  id: ID! @id @unique
  nameTwo: String
}

union Person = PersonOne | PersonTwo

type Movie {
  id: ID
  actors: [Person!]! @relationship(type: "ACTED_IN", direction: IN, nestedOperations: [CONNECT_OR_CREATE])
}
----
