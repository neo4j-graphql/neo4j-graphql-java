= @settable -> Relationships to a union type -> Prevent relationship field update

== Source schema

[source,graphql,schema=true]
----
type Movie @query(aggregate: true) {
  title: String!
  description: String
}

type Series @query(aggregate: true) {
  name: String!
  description: String
}

union Production = Movie | Series

type Actor @query(aggregate: true) {
  name: String!
  actedIn: [Production!]! @relationship(type: "ACTED_IN", direction: OUT) @settable(onCreate: true, onUpdate: false)
}
----
