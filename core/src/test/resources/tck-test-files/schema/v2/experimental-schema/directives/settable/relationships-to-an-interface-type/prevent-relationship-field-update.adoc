= @settable -> Relationships to an interface type -> Prevent relationship field update

== Source schema

[source,graphql,schema=true]
----
interface Production {
  title: String!
  description: String
}

type Movie implements Production @query(aggregate: true) {
  title: String!
  description: String
}

type Series implements Production @query(aggregate: true) {
  title: String!
  description: String
}

type Actor @query(aggregate: true) {
  name: String!
  actedIn: [Production!]! @relationship(type: "ACTED_IN", direction: OUT) @settable(onCreate: true, onUpdate: false)
}
----
