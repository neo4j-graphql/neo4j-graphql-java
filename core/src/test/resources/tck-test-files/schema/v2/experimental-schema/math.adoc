  viewers: Int!

  edges: [MovieEdge!]!
  pageInfo: PageInfo!
  totalCount: Int!
}

type PageInfo {
  endCursor: String
  hasNextPage: Boolean!
  hasPreviousPage: Boolean!
  startCursor: String
}

  movies(options: MovieOptions, where: MovieWhere): [Movie!]!
