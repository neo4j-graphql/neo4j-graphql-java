== Source schema

[source,graphql,schema=true]
----
interface Animal @plural(value: "animales") {
  name: String
}

type Dog implements Animal {
  name: String
  breed: String
}

type Cat {
  queenOf: String
}

union Pet @plural(value: "petties") = Dog | Cat
----
