type Movie {
  id: ID! @id @unique
  title: String!
  actors: [Actor!]! @relationship(type: "ACTED_IN", direction: IN)
}

type Actor {
  id: ID! @id @unique
  name: String!
  dob: DateTime!
  homeAddress: Point!
}
