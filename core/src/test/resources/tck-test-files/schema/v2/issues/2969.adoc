== authorAggregate should not be generated

.Schema
[source,graphql,schema=true]
----
type Post {
  content: String!
  author: User! @relationship(type: "HAS_AUTHOR", direction: OUT)
}

type User {
  id: ID!
  name: String!
  posts: [Post!]! @relationship(type: "HAS_AUTHOR", direction: IN)
}
----
