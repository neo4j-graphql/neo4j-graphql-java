== Source schema

[source,graphql,schema=true]
----
interface IProduct {
  id: String!
  name: String!
  genre: Genre!
}

type Movie implements IProduct {
  id: String!
  name: String!
  genre: Genre!
}

type Series implements IProduct {
  id: String!
  name: String!
  genre: Genre!
}

type Genre {
  name: String! @unique
  product: [IProduct!]! @relationship(type: "HAS_GENRE", direction: IN)
}
----
