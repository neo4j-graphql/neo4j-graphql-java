= Unions

== Unions

.Schema
[source,graphql,schema=true]
----
union Search = Movie | Genre

type Genre {
  id: ID
}

type Movie {
  id: ID
  search: [Search!]! @relationship(type: "SEARCH", direction: OUT)
  searchNoDirective: Search
}
----
