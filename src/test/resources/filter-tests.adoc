== Schema

[source,graphql,schema=true]
----
enum Gender { female, male }
type Person {
  id : ID!
  name: String
  age: Int
  height: Float
  fun: Boolean
  gender: Gender
  company: Company @relation(name:"WORKS_AT")
  location: _Neo4jPoint
}
type Company {
  _id: ID
  name: String
  employees: [Person] @relation(name:"WORKS_AT", direction: IN)
}
type Query {
  person : [Person]
}
----
